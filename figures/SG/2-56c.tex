% 定制画布
\makeatletter
\tikzoption{canvas is plane}[]{\@setOxy#1}
\def\@setOxy O(#1,#2,#3)x(#4,#5,#6)y(#7,#8,#9)%
  {\def\tikz@plane@origin{\pgfpointxyz{#1}{#2}{#3}}%
   \def\tikz@plane@x{\pgfpointxyz{#4}{#5}{#6}}%
   \def\tikz@plane@y{\pgfpointxyz{#7}{#8}{#9}}%
   \tikz@canvas@is@plane
  }
\makeatother

\begin{document}
\tdplotsetmaincoords{60}{140}
\begin{tikzpicture}[tdplot_main_coords,scale=1.1]
  % tikz 设定
  \pgfmathsetmacro{\final}{2.0*pi}
  \tikzset{Vertex/.style={circle, draw, fill=black, inner sep=0.8pt}}
  \tikzset{Dashed/.style={dash pattern = on 2.5pt off 1pt}}
  \tikzset{Dotted/.style={dash pattern = on 2.5pt off 0.8pt on 1pt off 0.6pt}}
  \foreach \angulo in {0,0.015,...,\final}{
    \pgfmathparse{cos(\angulo r)}
    \pgfmathsetmacro{\ppx}{\pgfmathresult}
    \pgfmathparse{sin(\angulo r)}
    \pgfmathsetmacro{\ppy}{\pgfmathresult}
    \begin{scope}[canvas is plane={O(0,0,0)x(\ppx,\ppy,0)y(0,0,1)}]
      \draw[very thin,darkgray,opacity=0.3](2,0)circle(0.4);
    \end{scope}
  }
  \begin{scope}[canvas is plane={O(0,0,0)x(0,1,0)y(0,0,1)}]
      \draw[darkgray](2,0)circle(0.4);
  \end{scope}
  \draw[Dotted,thin] (0,0,2)node[right]{$a$} -- (0, 0, -2);
\end{tikzpicture}
