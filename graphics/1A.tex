\documentclass{standalone}
\usepackage{tikz,amsmath}
\usepackage{tkz-euclide,ninecolors}
\begin{document}
\begin{tikzpicture}[>=latex, scale=1.5]
  \useasboundingbox(-4,-5)rectangle(2.5,5);
  \tkzDefPoint(0,0){O}
  \tkzDefPoint(1,0){a0}
  \tkzDrawSegment[blue](O,a0)
  \foreach \i [count=\j] in {0,...,10}
    {%
    \tkzDefPointWith[orthogonal normed](a\i,O)
    \tkzGetPoint{a\j}
    \tkzDefMidPoint(O,a\i)
    \tkzGetPoint{m\j}
    \tkzDrawPolySeg[color=blue](a\i,a\j,O)
    \node at (m\j){\small$\sqrt{\j}$};
  }
\end{tikzpicture}
\end{document}
